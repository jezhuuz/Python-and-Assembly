0 ***x -> S 0
0 ***S -> X 1
1 **x* -> W 1
1 **W* -> X 2
2 x*W* -> N 2
2 Nx** -> E 2
2 NE** -> X 3
3 ***x -> S 3
3 **xS -> W 3
3 *xWS -> X 4
4 **** -> N 5 #move up one block
5 *x** -> E 5 #moving east
5 *E** -> X 6
6 *E** -> N 7 #moving up one block
7 **x* -> W 7 #moving west
7 **W* -> X 4 #repeat the whole process until covered
